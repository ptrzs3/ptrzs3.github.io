Debug & Display Link to heading

两者除了特征名称不一样，要求实现的方法一模一样

pub trait Debug {
    // Required method
    fn fmt(&self, f: &mut Formatter<'_>) -> Result<(), Error>;
}

pub trait Display {
    // Required method
    fn fmt(&self, f: &mut Formatter<'_>) -> Result<(), Error>;
}

根据文档描述，Debug面向程序员，在生产环境中不应该出现；

为一个类型实现Display，会自动实现ToStringtrait，允许使用.to_string()方法。更应该为一个类型实现Display而不是ToString。

From & Into Link to heading 

struct A {
    member_A: u8,
    member_B: String,
    member_C: bool
}

struct B {
    member_A: char,
    member_B: f32
}

impl From<A> for B {
    fn from(value: Keyboard) -> Self {
        // details
    }
}

为B类型实现From trait，可以从A转换成B类型：let b:B = B::from(A)；实现Fromtrait后，Intotrait也自动实现：let b:B = a.into(B)，更推荐实现From而不是Into；但是并没有为A实现Fromtrait，因此let a:A = A::from(B)或者let a:A = b.into(A)是非法的。另外，使用into()方法时，必须指明接收值的类型。

迭代器与For循环 Link to heading

IntoIterator：实现该trait可以使用for循环，将隐式调用into_iter()方法将该类型转换为迭代器

Iterator：实现该特征即定义为迭代器，for循环就是不断调用该trait的next()方法

struct Iter<'a> {
    index: u8,
    songs: &'a u8
}
impl<'a> Iter<'a> {
    fn new(p: &'a PlayList) -> Self {
        Iter { songs: &p.songs, index: p.playing }
    }
}
impl<'a> Iterator for Iter<'a> {
    type Item = u8;
    fn next(&mut self) -> Option<Self::Item> {
        if self.index < *self.songs {
            self.index += 1;
            return Some(self.index);
        } 
        return None;
    }
}

struct IntoIter {
    playing: u8,
    songs: u8
}
impl IntoIter {
    fn new(p: PlayList) -> Self{
        IntoIter{
            playing: p.playing,
            songs: p.songs
        }
    }
}
impl Iterator for IntoIter {
    type Item = u8;
    fn next(&mut self) -> Option<Self::Item> {
        if self.playing < self.songs {
            self.playing += 1;
            return Some(self.playing);
        }
        return None;
    }
}

struct IterMut<'a> {
    playing: &'a mut u8,
    songs: &'a mut u8
}
impl<'a> IterMut<'a> {
    fn new(p: &'a mut PlayList) -> Self {
        IterMut { playing: &mut p.playing, songs: &mut p.songs }
    }
}
impl<'a> Iterator for IterMut<'a>{
    type Item = u8;
    fn next(&mut self) -> Option<Self::Item> {
        if self.playing < self.songs {
            *self.playing += 1;
            return Some(*self.playing)
        }
        return None;
    }
}

// PlayList定义
struct PlayList {
    playing: u8,
    songs: u8
}
impl PlayList {
    fn new() -> Self {
        PlayList {
            playing: 0,
            songs: 5
        }
    }
    fn iter(&self) -> Iter{
        Iter::new(self)
    }
    fn into_iter(self) -> IntoIter {
        IntoIter::new(self)
    }
    fn iter_mut(&mut self) -> IterMut {
        IterMut::new(self)
    }
}

// 借用
// 实现IntoIterator，可用for循环遍历
impl<'a> IntoIterator for &'a PlayList {
    type Item = u8;
    type IntoIter = Iter<'a>;
    fn into_iter(self) -> Self::IntoIter {
        self.iter()
    }
}
// 拿走所有权
impl IntoIterator for PlayList {
    type Item = u8;
    type IntoIter = IntoIter;
    fn into_iter(self) -> Self::IntoIter {
        self.into_iter()
    }
}
// 可变借用
impl<'a> IntoIterator for &'a mut PlayList {
    type Item = u8;
    type IntoIter = IterMut<'a>;
    fn into_iter(self) -> Self::IntoIter {
        self.iter_mut()
    }
}

fn main() {
    let mut playlist = PlayList::new();
    
    // 不可变借用
    for i in &playlist {
        println!("{i}");
    }

    // 可变借用
    for i in &mut playlist {
        println!("{i}");
    }

    // 取走所有权
    // 可变借用执行后，playing=songs，所以这个循环不输出内容
    for i in playlist {
        println!("{i}");
    }
}
